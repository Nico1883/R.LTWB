Calcular longitud serie CNE_IDEAM.shp en años LAnoSerie = ([FECHA_SUSP]-[FECHA_INST])/365

from datetime import datetime
def long_annos_serie(fecha_instalacion, fecha_suspension):
    if fecha_suspension:
        diff_date = datetime.strptime(fecha_suspension, '%Y-%m-%d') - datetime.strptime(fecha_instalacion, '%Y-%m-%d')
        return diff_date.days/365
    else:
        diff_date = datetime.strptime('2021-12-31', '%Y-%m-%d') - datetime.strptime(fecha_instalacion, '%Y-%m-%d')
        return diff_date.days/365
